```javascript
async function startScan() {
    // Step 1: submit 创建任务
    const submitResp = await fetch(`${API_URL}/api/scan/submit`, {...});
    currentTaskId = submitResp.data.task_id;
    
    // Step 2: execute 启动扫描（不再等待结果）
    fetch(`${API_URL}/api/scan/execute`, {
        method: 'POST',
        body: JSON.stringify({task_id, url, headers, param_name})
    }); // 不等待返回
    
    // Step 3: 开始轮询 progress
    startPolling(currentTaskId);
}

function startPolling(taskId) {
    pollTimer = setInterval(async () => {
        const resp = await fetch(`${API_URL}/api/scan/progress?task_id=${taskId}`);
        const data = await resp.json();
        
        if (data.code === 0) {
            updateProgressUI(data.data);
            
            if (data.data.status === 'completed') {
                clearInterval(pollTimer);
                loadResult(taskId);
            } else if (data.data.status === 'failed') {
                clearInterval(pollTimer);
                showError(data.data.error);
            }
        }
    }, 1000); // 每1秒轮询
}

    document.getElementById('progressFill').style.width = data.progress + '%';
    document.getElementById('progressText').textContent =

}
```

```python
from flask import Flask, request, jsonify
from flask_cors import CORS

app = Flask(__name__)
CORS(app)


# 检测器列表（定义步骤）
DETECTORS = [
    {"name": "提示词注入检测", "detector": "prompt_injection", "weight": 15},
    {"name": "越狱攻击检测", "detector": "jailbreak", "weight": 15},
    {"name": "数据泄露检测", "detector": "data_leak", "weight": 10},
    {"name": "权限绕过检测", "detector": "auth_bypass", "weight": 10},
    {"name": "隐私数据检测", "detector": "privacy", "weight": 10},
    {"name": "敏感内容检测", "detector": "sensitive", "weight": 10},
]


@app.route('/api/scan/execute', methods=['POST', 'GET', 'OPTIONS'])
def execute_scan():
    """执行扫描任务（分步更新进度）"""
    if request.method == 'OPTIONS':
        return jsonify({'code': 0}), 200

    if request.method == 'GET':
        return jsonify({
            'code': 0,
            'message': 'execute API is running',
            'detectors': [d['name'] for d in DETECTORS]
        })

    data = request.get_json()
    if not data:
        return jsonify({'code': 1, 'message': 'No JSON data provided'}), 400

    task_id = data.get('task_id')
    url = data.get('url')
    headers = data.get('headers', {})
    param_name = data.get('param_name', 'query')

    if not task_id or not url:
        return jsonify({'code': 1, 'message': 'task_id and url required'}), 400

    # 导入依赖
    from lib.db import get_session, ScanTask, Vulnerability, init_db
    from scanner.engine import ScanEngine

    # 初始化数据库
    init_db()
    db = get_session()

    if not db:
        return jsonify({'code': 1, 'message': 'Database connection failed'}), 500

    try:
        # ===== 步骤 1: 初始化 (progress: 0-10) =====
        task = db.query(ScanTask).filter(ScanTask.id == task_id).first()
        if not task:
            return jsonify({'code': 1, 'message': 'Task not found'}), 404

        task.status = 'running'
        task.progress = 5
        task.current_step = '初始化扫描引擎'
        db.commit()

        # ===== 步骤 2: 加载规则 (progress: 10-20) =====
        task.progress = 10
        task.current_step = '加载检测规则'
        db.commit()

        engine = ScanEngine()
        engine.load_rules()

        task.progress = 20
        db.commit()

        # ===== 步骤 3: 执行检测器 (progress: 20-80) =====
        all_vulnerabilities = []
        total_weight = sum(d['weight'] for d in DETECTORS)
        progress_per_detector = 60 / len(DETECTORS)  # 60% 分配给检测器

        for i, detector_info in enumerate(DETECTORS):
            detector_name = detector_info['name']
            detector_key = detector_info['detector']

            # 更新进度
            task.current_step = detector_name
            task.progress = 20 + int(i * progress_per_detector)
            db.commit()

            # 执行检测
            try:
                detector = engine.get_detector(detector_key)
                if detector:
                    vulns = detector.detect(url, headers, param_name)
                    all_vulnerabilities.extend(vulns)
            except Exception as e:
                print(f"Detector {detector_name} failed: {e}")
                # 继续执行其他检测器

            # 完成该检测器
            task.progress = 20 + int((i + 1) * progress_per_detector)
            db.commit()

        # ===== 步骤 4: 计算评分 (progress: 80-90) =====
        task.current_step = '计算安全评分'
        task.progress = 80
        db.commit()

        from scanner.scorer import Scorer
        scorer = Scorer()
        score = scorer.calculate(all_vulnerabilities)
        level = scorer.get_level(score)

        # ===== 步骤 5: 保存漏洞 (progress: 90-95) =====
        task.current_step = '保存检测结果'
        task.progress = 90
        db.commit()

        for i, vuln in enumerate(all_vulnerabilities):
            vuln_id = f"{task_id}_{vuln.get('rule_id', 'unknown')}_{i}_{uuid.uuid4().hex[:4]}"
            vuln_record = Vulnerability(
                id=vuln_id,
                task_id=task_id,
                rule_id=vuln.get('rule_id', ''),
                name=vuln.get('name', ''),
                severity=vuln.get('severity', 'medium'),
                score_deduction=vuln.get('score_deduction', 0),
                description=vuln.get('description', ''),
                suggestion=vuln.get('suggestion', ''),
                evidence=json.dumps(vuln.get('evidence', [])),
            )
            db.add(vuln_record)

        db.commit()

        # ===== 步骤 6: 完成 (progress: 100) =====
        task.status = 'completed'
        task.progress = 100
        task.current_step = '扫描完成'
        task.score = score
        task.level = level
        task.completed_at = datetime.now()
        db.commit()
        db.close()

        return jsonify({
            'code': 0,
            'message': 'success',
            'data': {
                'task_id': task_id,
                'score': score,
                'level': level,
                'vulnerabilities': all_vulnerabilities,
            }
        })

    except Exception as e:
        print(f"Execute scan error: {e}")
        
        # 更新失败状态
        task = db.query(ScanTask).filter(ScanTask.id == task_id).first()
        if task:
            task.status = 'failed'
            task.error_message = str(e)
            db.commit()
        
        db.close()
        
        return jsonify({
            'code': 1,
            'message': f'扫描执行失败: {str(e)}',
            'data': {
                'task_id': task_id,
                'status': 'failed',
                'error': str(e)
            }
        })
```

```python
"""
扫描引擎 - 增强错误边界
"""

class ScanEngine:
    
    def __init__(self):
        self.detectors = {}
        self.rules_loaded = False
    
    def load_rules(self):
        """加载规则（带错误处理）"""
        try:
            from scanner.rules_loader import RulesLoader
            loader = RulesLoader()
            self.rules = loader.load_all()
            self.rules_loaded = True
        except Exception as e:
            print(f"Rules load failed: {e}")
            self.rules = []
            self.rules_loaded = False
    
    def get_detector(self, detector_key):
        """获取检测器（带错误处理）"""
        if detector_key in self.detectors:
            return self.detectors[detector_key]
        
        try:
            if detector_key == 'prompt_injection':
                from scanner.detectors.prompt_injection import PromptInjectionDetector
                self.detectors[detector_key] = PromptInjectionDetector()
            elif detector_key == 'jailbreak':
                from scanner.detectors.jailbreak import JailbreakDetector
                self.detectors[detector_key] = JailbreakDetector()
            elif detector_key == 'data_leak':
                from scanner.detectors.data_leak import DataLeakDetector
                self.detectors[detector_key] = DataLeakDetector()
            elif detector_key == 'auth_bypass':
                from scanner.detectors.auth_bypass import AuthBypassDetector
                self.detectors[detector_key] = AuthBypassDetector()
            elif detector_key == 'privacy':
                from scanner.detectors.privacy import PrivacyDetector
                self.detectors[detector_key] = PrivacyDetector()
            elif detector_key == 'sensitive':
                from scanner.detectors.sensitive import SensitiveDetector
                self.detectors[detector_key] = SensitiveDetector()
            
            return self.detectors.get(detector_key)
        except Exception as e:
            print(f"Detector {detector_key} load failed: {e}")
            return None
    
    def scan(self, task_id, url, headers, param_name):
        """
        执行扫描（单个检测器失败不影响整体）
        
        Returns:
            ScanResult with vulnerabilities
        """
        all_vulnerabilities = []
        failed_detectors = []
        
        detector_keys = ['prompt_injection', 'jailbreak', 'data_leak', 
                         'auth_bypass', 'privacy', 'sensitive']
        
        for key in detector_keys:
            detector = self.get_detector(key)
            if detector is None:
                failed_detectors.append(key)
                continue
            
            try:
                vulns = detector.detect(url, headers, param_name)
                if vulns:
                    all_vulnerabilities.extend(vulns)
            except Exception as e:
                print(f"Detector {key} execution failed: {e}")
                failed_detectors.append(key)
                # 继续执行下一个检测器
        
        # 记录失败的检测器（但不影响整体结果）
        if failed_detectors:
            print(f"Warning: {len(failed_detectors)} detectors failed: {failed_detectors}")
        
        return {
            'vulnerabilities': all_vulnerabilities,
            'failed_detectors': failed_detectors
        }
```
