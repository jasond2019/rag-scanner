#### 3. test_scan_flow_live.py 示例

```python
"""
真实 API 流程测试（调用 Vercel）
"""
import pytest
import requests

API_URL = "https://rag-scanner.vercel.app"


@pytest.mark.skipif(not os.environ.get("TEST_LIVE"), reason="Live test disabled")
class TestLiveScanFlow:

    def test_complete_scan_flow(self):
        """完整扫描流程"""
        # 1. 提交
        resp = requests.post(f"{API_URL}/api/scan/submit", json={
            "url": "https://httpbin.org/post",
            "input_type": "url",
            "user_id": "pytest_live"
        })
        assert resp.status_code == 200
        task_id = resp.json()["data"]["task_id"]

        # 2. 执行
        resp = requests.post(f"{API_URL}/api/scan/execute", json={
            "task_id": task_id,
            "url": "https://httpbin.org/post",
            "headers": {},
            "param_name": "query"
        })
        assert resp.status_code == 200

        # 3. 查询进度
        resp = requests.get(f"{API_URL}/api/scan/progress?task_id={task_id}")
        assert resp.status_code == 200

        # 4. 获取结果
        resp = requests.get(f"{API_URL}/api/scan/result?task_id={task_id}")
        data = resp.json()
        assert data["code"] == 0
        assert "score" in data["data"]
        assert "level" in data["data"]
```
